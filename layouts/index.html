Blazingly Fast

Actix is blazingly fast. Check yourself.

extern crate actix_web;
use actix_web::{server, App, HttpRequest};

fn index(req: HttpRequest) -> &'static str {
    "Hello world!"
}

fn main() {
    server::new(
        || App::new()
            .resource("/", |r| r.f(index)))
        .bind("127.0.0.1:8088")
            .expect("Can not bind to 127.0.0.1:8088")
        .run();
}
